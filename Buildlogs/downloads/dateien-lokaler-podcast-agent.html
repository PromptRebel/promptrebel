def write_status(data: dict) -> None:
    STATUS_FILE.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")


def read_status() -> dict:
    if not STATUS_FILE.exists():
        return {"state": "idle"}
    try:
        return json.loads(STATUS_FILE.read_text(encoding="utf-8"))
    except Exception:
        return {"state": "error", "message": "status.json konnte nicht gelesen werden."}


@app.route("/", methods=["GET"])
def index():
    return HTML


@app.route("/start", methods=["POST"])
def start():
    data = request.get_json(silent=True) or {}
    thema = (data.get("thema") or "KI im Alltag").strip()

    current = read_status()
    if current.get("state") == "running":
        return jsonify({"ok": False, "message": "Es läuft bereits ein Job."}), 409

    write_status({
        "state": "running",
        "topic": thema,
        "stage": "Start",
        "progress": "0%",
        "message": "Job wurde gestartet."
    })

    subprocess.Popen([
        str(PYTHON_EXE),
        str(BASE_DIR / "generate_text.py"),
        thema
    ])

    return jsonify({"ok": True})


@app.route("/status", methods=["GET"])
def status():
    return jsonify(read_status())


@app.route("/latest-audio", methods=["GET"])
def latest_audio():
    import glob
    from flask import send_file

    wavs = sorted(glob.glob(str(BASE_DIR / "podcast_*.wav")))
    if not wavs:
        return ("Keine Audio-Datei gefunden.", 404)

    latest = max(wavs, key=lambda p: Path(p).stat().st_mtime)
    return send_file(latest, mimetype="audio/wav")


if __name__ == "__main__":
    app.run(host="0.0.0.0", port=3000)

import datetime
import subprocess
import sys
import time
import json
from pathlib import Path

BASE_DIR = Path(r"C:\xtts-clean")
OUTPUT_DIR = BASE_DIR
OLLAMA_MODEL = "llama3"
STATUS_FILE = BASE_DIR / "status.json"


def write_status(data: dict) -> None:
    STATUS_FILE.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")


def get_topic() -> str:
    if len(sys.argv) > 1:
        return " ".join(sys.argv[1:]).strip()
    return "KI im Alltag"


def generate_text(topic: str) -> str:
    prompt = f"""
Schreibe einen natürlich gesprochenen Podcast-Monolog auf Deutsch zum Thema: {topic}.

Regeln:
- genau ein Sprecher
- 8 bis 10 kurze Sätze
- maximal 140 Wörter
- nur deutsche Wörter
- keine englischen Wörter
- keine fremdsprachigen Zeichen
- keine Überschrift
- keine Rollenbezeichnungen
- keine Aufzählungen
- keine Klammern
- keine Emojis
- gut vorlesbar
- sachlich und klar
""".strip()

    result = subprocess.run(
        ["ollama", "run", OLLAMA_MODEL],
        input=prompt,
        text=True,
        capture_output=True,
        encoding="utf-8",
        errors="ignore",
        check=False,
    )

    if result.returncode != 0:
        raise RuntimeError(f"Ollama-Fehler:\\n{result.stderr}")

    text = " ".join(result.stdout.strip().split())
    if not text:
        raise RuntimeError("Ollama hat keinen Text zurückgegeben.")

    return text


def main() -> None:
    topic = get_topic()
    timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")

    write_status({
        "state": "running",
        "topic": topic,
        "stage": "Textgenerierung",
        "progress": "10%",
        "message": "Ollama erzeugt den Text."
    })

    text = generate_text(topic)

    txt_path = OUTPUT_DIR / f"podcast_{timestamp}.txt"
    txt_path.write_text(text, encoding="utf-8")

    write_status({
        "state": "running",
        "topic": topic,
        "stage": "Text gespeichert",
        "progress": "35%",
        "message": f"TXT gespeichert: {txt_path.name}"
    })

    subprocess.run(["taskkill", "/IM", "ollama.exe", "/F"], capture_output=True)

    write_status({
        "state": "running",
        "topic": topic,
        "stage": "RAM freigeben",
        "progress": "45%",
        "message": "Ollama wurde beendet."
    })

    time.sleep(3)

    write_status({
        "state": "running",
        "topic": topic,
        "stage": "Vertonung startet",
        "progress": "50%",
        "message": "speak_text.py wird gestartet."
    })

    subprocess.run([
        str(BASE_DIR / ".venv" / "Scripts" / "python.exe"),
        str(BASE_DIR / "speak_text.py"),
        str(txt_path)
    ], check=True)


if __name__ == "__main__":
    try:
        main()
    except Exception as e:
        write_status({
            "state": "error",
            "stage": "generate_text.py",
            "message": str(e)
        })
        raise

speak_text.py
Speicherort: C:\xtts-clean\speak_text.py
Zweck: Vertont den Text per XTTS, schreibt Fortschritt in status.json und baut die WAV-Datei zusammen
import gc
import json
import re
import shutil
import subprocess
import sys
import time
import wave
from pathlib import Path

BASE_DIR = Path(r"C:\xtts-clean")
TTS_EXE = BASE_DIR / ".venv" / "Scripts" / "tts.exe"
VOICE_WAV = BASE_DIR / "voice.wav"
OUTPUT_DIR = BASE_DIR
SEGMENTS_DIR = BASE_DIR / "segments"
STATUS_FILE = BASE_DIR / "status.json"

TTS_MODEL = "tts_models/multilingual/multi-dataset/xtts_v2"
LANGUAGE = "de"

MAX_SENTENCES = 8
MAX_CHARS_PER_SENTENCE = 140


def write_status(data: dict) -> None:
    STATUS_FILE.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")


def get_latest_txt() -> Path:
    txt_files = sorted(
        BASE_DIR.glob("podcast_*.txt"),
        key=lambda p: p.stat().st_mtime,
        reverse=True
    )
    if not txt_files:
        raise FileNotFoundError("Keine podcast_*.txt Datei gefunden.")
    return txt_files[0]


def split_into_sentences(text: str) -> list[str]:
    parts = re.split(r"(?<=[.!?])\s+", text)
    sentences = []

    for part in parts:
        sentence = " ".join(part.split()).strip()
        if not sentence:
            continue

        if len(sentence) > MAX_CHARS_PER_SENTENCE:
            chunks = re.findall(rf".{{1,{MAX_CHARS_PER_SENTENCE}}}(?:\s|$)", sentence)
            for chunk in chunks:
                chunk = chunk.strip()
                if chunk:
                    sentences.append(chunk)
        else:
            sentences.append(sentence)

    return sentences[:MAX_SENTENCES]


def synthesize_sentence(sentence: str, out_file: Path) -> None:
    cmd = [
        str(TTS_EXE),
        "--model_name", TTS_MODEL,
        "--text", sentence,
        "--language_idx", LANGUAGE,
        "--speaker_wav", str(VOICE_WAV),
        "--out_path", str(out_file),
    ]

    result = subprocess.run(
        cmd,
        text=True,
        capture_output=True,
        encoding="utf-8",
        errors="ignore",
        check=False,
    )

    if result.returncode != 0:
        raise RuntimeError(
            f"XTTS-Fehler bei {out_file.name}\\n\\n"
            f"Returncode: {result.returncode}\\n\\n"
            f"TEXT: {sentence}\\n\\n"
            f"STDERR:\\n{result.stderr}\\n\\n"
            f"STDOUT:\\n{result.stdout}"
        )

    if not out_file.exists():
        raise RuntimeError(f"Keine WAV erzeugt: {out_file}")

    if out_file.stat().st_size == 0:
        raise RuntimeError(f"WAV ist leer: {out_file}")


def concatenate_wavs(segment_files: list[Path], final_output: Path) -> None:
    if not segment_files:
        raise RuntimeError("Keine WAV-Segmente zum Zusammenfügen vorhanden.")

    with wave.open(str(segment_files[0]), "rb") as first:
        params = first.getparams()
        frames = [first.readframes(first.getnframes())]

    for wav_file in segment_files[1:]:
        with wave.open(str(wav_file), "rb") as w:
            current_params = w.getparams()

            if (
                current_params.nchannels != params.nchannels
                or current_params.sampwidth != params.sampwidth
                or current_params.framerate != params.framerate
                or current_params.comptype != params.comptype
            ):
                raise RuntimeError(
                    f"WAV-Parameter passen nicht zusammen: {wav_file.name}\\n"
                    f"Erwartet: channels={params.nchannels}, width={params.sampwidth}, "
                    f"rate={params.framerate}, comptype={params.comptype}\\n"
                    f"Gefunden: channels={current_params.nchannels}, width={current_params.sampwidth}, "
                    f"rate={current_params.framerate}, comptype={current_params.comptype}"
                )

            frames.append(w.readframes(w.getnframes()))

    with wave.open(str(final_output), "wb") as out:
        out.setparams(params)
        for data in frames:
            out.writeframes(data)


def main() -> None:
    if not TTS_EXE.exists():
        raise FileNotFoundError(f"TTS nicht gefunden: {TTS_EXE}")
    if not VOICE_WAV.exists():
        raise FileNotFoundError(f"voice.wav nicht gefunden: {VOICE_WAV}")

    txt_path = Path(sys.argv[1]) if len(sys.argv) > 1 else get_latest_txt()
    if not txt_path.exists():
        raise FileNotFoundError(f"TXT-Datei nicht gefunden: {txt_path}")

    text = txt_path.read_text(encoding="utf-8").strip()
    if not text:
        raise RuntimeError("TXT-Datei ist leer.")

    timestamp = txt_path.stem.replace("podcast_", "")
    final_output = OUTPUT_DIR / f"podcast_{timestamp}.wav"

    if SEGMENTS_DIR.exists():
        shutil.rmtree(SEGMENTS_DIR)
    SEGMENTS_DIR.mkdir(parents=True, exist_ok=True)

    sentences = split_into_sentences(text)
    if not sentences:
        raise RuntimeError("Keine Sätze zum Vertonen gefunden.")

    gc.collect()
    time.sleep(5)

    total = len(sentences)
    segment_files = []

    for i, sentence in enumerate(sentences, start=1):
        progress = 50 + int((i / total) * 40)

        write_status({
            "state": "running",
            "stage": "Vertonung",
            "progress": f"{progress}%",
            "message": f"Segment {i}/{total}",
        })

        segment_path = SEGMENTS_DIR / f"segment_{i:03d}.wav"
        synthesize_sentence(sentence, segment_path)
        segment_files.append(segment_path)

        gc.collect()
        time.sleep(1)

    write_status({
        "state": "running",
        "stage": "Zusammenfügen",
        "progress": "95%",
        "message": "Segmente werden zusammengebaut."
    })

    concatenate_wavs(segment_files, final_output)

    if not final_output.exists():
        raise RuntimeError("Finale WAV wurde nicht erstellt.")

    shutil.rmtree(SEGMENTS_DIR, ignore_errors=True)

    write_status({
        "state": "done",
        "stage": "Fertig",
        "progress": "100%",
        "message": "Podcast wurde erfolgreich erstellt.",
        "output_file": final_output.name,
        "output_url": "/latest-audio"
    })


if __name__ == "__main__":
    try:
        main()
    except Exception as e:
        write_status({
            "state": "error",
            "stage": "speak_text.py",
            "message": str(e)
        })
        raise
